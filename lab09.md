```mysql
DELIMITER //
CREATE TRIGGER kreatura_before_insert 
BEFORE INSERT ON kreatura
FOR EACH ROW 
BEGIN 
	IF NEW.waga<=0
	THEN 
		SET NEW.waga=1;
	END IF;
END
//
```
